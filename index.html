// 在滑動判斷邏輯中注入防護 function handleTouchMove(e) { const isMemoryPage = (currentIndex === 3); // 假設回憶頁是第 4 頁 const memoryArea = document.getElementById('memoryArea'); if (isMemoryPage) { // 如果回憶列表還沒捲動到頂部或底部，禁止 App 大換頁 const isAtTop = memoryArea.scrollTop === 0; const isAtBottom = memoryArea.scrollHeight - memoryArea.scrollTop === memoryArea.clientHeight; if (!isAtTop && !isAtBottom) { e.stopPropagation(); // 阻止事件向上傳遞，鎖定在回憶列表內 return; } } // 執行原有的換頁滑動判斷... }
